import pandas as pd
import numpy as np
import re
from collections import Counter
import string
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize
from nltk.stem import WordNetLemmatizer
import matplotlib.pyplot as plt

# Load and preprocess the data
df = pd.read_csv('data_feedback.csv', encoding='latin1')
df['Text'] = df['Review Text'].astype(str)

lemmatizer = WordNetLemmatizer()
stopword_set = set(stopwords.words('english'))

def preprocess_text(text):
    text = text.lower()
    tokens = word_tokenize(text)
    tokens = [lemmatizer.lemmatize(word) for word in tokens if word not in stopword_set]
    return tokens

def clean_text(text):
    text = re.sub(r'http\S+', '', text)
    text = re.sub(r'[^\x00-\x7F]+', '', text)
    text = re.sub(f'[{re.escape(string.punctuation.replace(".", ""))}]', '', text)
    text = re.sub(r'\b\d+\b', '', text)
    text = re.sub(r'\.{2,}', '', text)
    return text

def remove_repeating_pattern(text):
    pattern = re.compile(r'\b(\w+\s?)(\.\s?\1)(2,)\b')
    return re.sub(pattern, '', text)

def remove_repeating_pattern_2(text):
    pattern = re.compile(r'\b(\w+)\s+\1(?:\s+\1)+\b', re.IGNORECASE)
    def remove_repeats(match):
        return match.group(1)
    return pattern.sub(remove_repeats, text)

# Apply cleaning functions
df["cleaned_data"] = df["Text"].apply(remove_repeating_pattern)
df["cleaned_data"] = df["cleaned_data"].apply(remove_repeating_pattern_2)
df["clean_text"] = df["cleaned_data"].apply(clean_text)
df["clean_text"] = df["clean_text"].apply(remove_repeating_pattern)

# Tokenize and lemmatize
df["tokens"] = df["clean_text"].apply(preprocess_text)

# Combine all tokens into a single list
all_tokens = [token for sublist in df["tokens"] for token in sublist]

# Calculate word frequencies
word_freq = Counter(all_tokens)

# Sort words by frequency and assign ranks
sorted_words = sorted(word_freq.items(), key=lambda item: item[1], reverse=True)
ranked_words = [(rank + 1, word, freq) for rank, (word, freq) in enumerate(sorted_words)]

# Convert to DataFrame for easy plotting
rank_freq_df = pd.DataFrame(ranked_words, columns=['Rank', 'Word', 'Frequency'])

# Find the h-point
h_point = next((rank for rank, word, freq in ranked_words if rank == freq), None)

if h_point is not None:
    print(f"H-Point: {h_point}")

    # Calculate text length N
    N = sum(word_freq.values())

    # Calculate a-point
    a_point = N / (h_point ** 2)
    print(f"A-Point: {a_point}")

    # Retrieve the 355th word and its frequency
    word_355 = rank_freq_df.loc[rank_freq_df['Rank'] == 355, 'Word'].values[0]
    freq_355 = rank_freq_df.loc[rank_freq_df['Rank'] == 355, 'Frequency'].values[0]
    print(f"The 355th word is '{word_355}' with a frequency of {freq_355}.")

    # Plot Rank-Frequency distribution
    plt.figure(figsize=(10, 6))
    plt.plot(rank_freq_df['Rank'], rank_freq_df['Frequency'], label='Rank-Frequency Distribution')
    plt.axvline(x=h_point, color='r', linestyle='--', label=f'H-Point: {h_point}')
    plt.axhline(y=h_point, color='r', linestyle='--')
    plt.scatter(h_point, h_point, color='red')

    # Annotate the h-point
    plt.text(h_point, h_point, f'  H-Point ({h_point}, {h_point})', horizontalalignment='left', color='red')

    # Plot the curve N = ah^2
    h_values = np.arange(1, max(rank_freq_df['Rank']) + 1)
    a_curve = a_point * (h_values ** 2)

    plt.plot(h_values, a_curve, label=f'N = a * h^2 (a = {a_point:.2f})', linestyle='--', color='blue')

    # Highlight the 355th word
    plt.scatter(355, freq_355, color='green')
    plt.text(355, freq_355, f'  355th Word ({355}, {freq_355})', horizontalalignment='left', color='green')

    plt.title('Rank-Frequency Distribution with H-Point and A-Curve')
    plt.xlabel('Rank')
    plt.ylabel('Frequency')
    plt.legend()
    plt.grid(True)

    # Save and show the plot
    plt.savefig('rank_frequency_distribution.png')
    plt.show()

else:
    print("H-Point not found in the data.")

    text = re.sub(r'http\S+', '', text)
    text = re.sub(r'[^\x00-\x7F]+', '', text)
    text = re.sub(f'[{re.escape(string.punctuation.replace(".", ""))}]', '', text)
    text = re.sub(r'\b\d+\b', '', text)
    text = re.sub(r'\.{2,}', '', text)
    return text

def remove_repeating_pattern(text):
    pattern = re.compile(r'\b(\w+\s?)(\.\s?\1)(2,)\b')
    return re.sub(pattern, '', text)

def remove_repeating_pattern_2(text):
    pattern = re.compile(r'\b(\w+)\s+\1(?:\s+\1)+\b', re.IGNORECASE)
    def remove_repeats(match):
        return match.group(1)
    return pattern.sub(remove_repeats, text)

# Apply cleaning functions
df["cleaned_data"] = df["Text"].apply(remove_repeating_pattern)
df["cleaned_data"] = df["cleaned_data"].apply(remove_repeating_pattern_2)
df["clean_text"] = df["cleaned_data"].apply(clean_text)
df["clean_text"] = df["clean_text"].apply(remove_repeating_pattern)

# Tokenize and lemmatize
df["tokens"] = df["clean_text"].apply(preprocess_text)

# Function to calculate sentiment score using TextBlob
def sentiment_score(text):
    return TextBlob(text).sentiment.polarity

# Calculate sentiment score for each feedback
df["sentiment_score"] = df["clean_text"].apply(sentiment_score)

# Define alpha and beta
alpha = 0.5
beta = 1.0

# Calculate frequencies of synthetic and analytic words
synthetic_words = []
analytic_words = []

for word in df["tokens"].explode():
    if len(word) > 6:  # Example criterion for synthetic words (could be refined)
        synthetic_words.append(word)
    else:
        analytic_words.append(word)

synthetic_freq = Counter(synthetic_words)
analytic_freq = Counter(analytic_words)

# Function to calculate enhanced score
def enhanced_score(text, alpha, beta):
    tokens = preprocess_text(text)
    synthetic_count = sum(1 for word in tokens if len(word) > 6)
    analytic_count = sum(1 for word in tokens if len(word) <= 6)
    
    if synthetic_count == analytic_count:  # Prevent division by zero
        freq_ratio = 0
    else:
        freq_ratio = (synthetic_count + analytic_count) / (synthetic_count - analytic_count)
    
    sentiment = sentiment_score(text)
    return alpha * sentiment + beta * freq_ratio

# Calculate enhanced score for each feedback
